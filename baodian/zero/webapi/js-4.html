  模拟一个瀑布流，点击增加一个a标签，我们可以通过某种方式让它无限往下加载，所以说div里面你可能根本就不知道到底有多少个a标签，你没法一个个的对a标签绑定一个click事件，所以说这个时候我们就能把事件绑定到div1上，这个时候每点击每个a标签都会通过冒泡机制冒到div1上面来。然后我们就可以通过某些手段去拿到这个a标签的点击，这就是事件代理的应用

  1. 代码简洁
  2. 减少浏览器的占用
  3. 但是，不要滥用

# 编写一个通用的事件监听函数

function bindEvent(elem, type, selector, fn) {
    if (fn == null) {
        fn = selector
        selector = null
    }
    elem.addEventListener(type, event => {
        const target = event.target
        if (selector) {
            // 代理绑定
            if (target.matches(selector)) { // 判断元素符不符合css选择器
                fn.call(target, event)
            }
        } else {
            // 普通绑定
            fn.call(target, event)
        }
    })
}


// 普通绑定
const a = document.getElementById('link1')
bindEvent(a, 'click', function (event) {
    // console.log(e.target) // 获取触发点击事件的元素
    event.preventDefault(); // 阻止默认行为
    alert(this.innerHTML)
})

// 代理绑定
const div1 = document.getElementById("div1")
bindEvent(div1, "click", "a", function (event) {
    e.preventDefault()
    alert(this.innerHTML)
})
