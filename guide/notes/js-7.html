    # call 和 apply 的实现

    声明在前的一句话：

    关于 call / apply 的实现，网上大神们的源码实现，深度解析都非常全面了，那么为什么我还要在写一篇这样的帖子呢？

    1，希望能够用更简单和基础的言语让初学的小伙伴们更好的理解。

    2，也是为了给自己的知识点做一个整理和记录。

    那么正题开始：

    要实现一个 API，首先得了解这个 API 的作用：MDN 对 call 的讲解和示例

    # call 实现的第一步：

    
首先看一个小例子：

var name = 'window'

function fn() {
    console.log( this.name )
}

var obj = {
    name: 'obj'
}

fn.call(obj) // 打印结果 --> obj


基于上面的小例子，我们知道了 call 最基本的作用：改变函数内部的 this 指向，并同时执行该函数。
也就是说上面 fn.call(obj) 等同于 obj.fn()
那么我们知道了原理，就可以开始动手了：

Function.prototype.myCall = function (obj) {
    // 参数 obj：最终 this 的指向将改变成 obj

    // 因为传入的 obj 是一个对象，对象可以添加属性，那么我们新增一个属性 fn
    // 这里的 this，就是 call 前面的函数，等同于 obj.fn = function fn(){ ... }
    obj.fn = this

    // obj 调用完该函数后，删除 fn 属性
    obj.fn()
    delete obj.fn
}

那么试试我们自己的 myCall 看看答案是否一致
fn.myCall(obj) // 打印结果 --> obj

    # call 实现的第二步：

    var name = 'window'

function fn(num1, num2) {
    console.log( this.name )
    console.log( num1 )
    console.log( num2 )
}

var obj = {
    name: 'obj'
}

fn.call(obj, 10, 20) // 打印结果 --> obj, 10, 20


那么问题来了，call传入的参数数量是不固定的。那么如何解决呢？没错，实参列表 arguments

Function.prototype.myCall = function (obj) {
    obj.fn = this

    var args = [] // 定义一个空数组，接收传入的参数

    // 注意，这里的 i 是从第 1 位开始循环的，因为第 0 个参数是 this 指向。
    // 我们真正需要用到的参数是从第 1 位开始的。
    // arguments 之所以能用 for 循环，是因为它是类数组，详情可以看个人以前的文章
    for(var i = 1; i < arguments.length; i++) {

        // 很多小伙伴们这里可能会看不懂
        // args 实际存储的是 [ 'arguments[1]', 'arguments[2]' ]
        args.push('arguments[' + i + ']')

    }

    // 那么这里的 eval 又是个什么操作呢？
    // 简单来说，就是用 JS 引擎来解析这一堆字符串里面的内容。
    // 你可以把 eval 看成是 <script> 标签
    // 例如：eval('function fn(a,b,c){console.log(a,b,c)}; fn(1,2,3)')
    // 等同于 <script> function fn(a,b,c){console.log(a,b,c)}; fn(1,2,3) </script>

    // 那么我们继续看个小例子：数组 + 字符串 = ？
    // var arr = [123, 456, 789]
    // console.log( arr + '' ) --> '123,456,789'

    // 那么这里的意思就等同于 obj.fn( arguments[1], arguments[2] )
    eval( 'obj.fn(' + args + ')' )

    delete obj.fn
}

那么接下来，再试试我们的 myCall
fn.myCall(obj, 10, 20) // 打印结果 --> obj, 10, 20

    # call 实现的第三步：

    我们知道除了上面之外：
call 的第一个参数可以为空，也可以是 null 还有 undefined，那么此时的 this 就指向 window。
以及 fn.call(obj) 执行完之后，也是可以有返回值的。

知道了这些，那么 myCall 就彻底完善了

Function.prototype.myCall = function (obj) {

    obj = obj || window // 如果没传参数，或者是 null 和 undefined，则赋值 window

    obj.fn = this

    var args = []

    for(var i = 1; i < arguments.length; i++) {
        args.push('arguments[' + i + ']')
    }

    var result = eval( 'obj.fn(' + args + ')' ) // 将 obj.fn() 执行后的返回值赋值给result

    delete obj.fn

    return result // 最后返回 result
}


至此，call 的实现就完成了，当然，也可以用 ES6 的方式更简单的实现。
了解了 call 的原理，小伙伴们就自己试试吧。

    关于 apply，其原理和 call 大同小异，区别就在于传参，这里就直接上成品代码了。

    Function.prototype.myApply = function (obj, arr) {
  var obj = obj || window;

  obj.fn = this;

  var result;

  if (!arr) {
    result = obj.fn(); // 没有传第二个参数的情况下，就直接 obj.fn() 执行
  } else {
    var args = [];

    // 这里的 i 从 0 开始循环，因为遍历的是第二个参数 arr
    for (var i = 0; i < arr.length; i++) {
      args.push("arr[" + i + "]");
    }

    result = eval("obj.fn(" + args + ")");
  }

  delete obj.fn;

  return result;
};
